TXBaseControl ::=   TEXTUAL-CONVENTION
    STATUS      current
    DESCRIPTION "basic control commands
                 0        reserved
                 600..699 reserved "
    SYNTAX INTEGER {
        on                  (1),
        off                 (2),
        totalPowerRaise     (3),
        totalPowerLower     (4)        
    }    

TXBaseStatus ::=   TEXTUAL-CONVENTION
    STATUS      current
    DESCRIPTION "Assigned to status"
    SYNTAX INTEGER {
        on                  (1),
        off                 (2)
    }    

SwitchOpt ::=   TEXTUAL-CONVENTION
    STATUS      current
    DESCRIPTION "switch exciter"
    SYNTAX INTEGER {
        excAOn                 (1),
        excBOn                 (2)
    }  

SwitchMode ::=   TEXTUAL-CONVENTION
    STATUS      current
    DESCRIPTION "switch exciter"
    SYNTAX INTEGER {
        auto                   (1),
        manual                 (2)
    }

AuxControlValue ::=   TEXTUAL-CONVENTION
    STATUS      current
    DESCRIPTION "switch exciter"
    SYNTAX INTEGER {
        auxOn                  (1),
        auxOff                 (2)
    }
